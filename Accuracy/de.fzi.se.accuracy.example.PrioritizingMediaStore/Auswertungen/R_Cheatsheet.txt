aia_ecdf_pdf <- function(main="Title", ...) {
	pdf(paste(main,".pdf"))
	aia_ecdf(...)
	dev.off()
}

aia_ecdf <- function(minResults, asSpecResults, maxResults, main="Cumulated Response Time", xlab="Response Time") {
	minX <- min(min(minResults),min(asSpecResults),min(maxResults));
	maxX <- max(max(minResults),max(asSpecResults),max(maxResults));
	plot.ecdf(minResults, xlim=c(minX, maxX), pch=3, col="green", main=main, xlab=xlab)
	plot.ecdf(asSpecResults, xlim=c(minX, maxX), pch=18, col="black", add=TRUE)
	plot.ecdf(maxResults, xlim=c(minX, maxX), pch=4, col="red", add=TRUE)
	legend("bottomright", c("Minimum","As Specified", "Maximum"), pch=c(3,18,4), title="Legend", col=c("green","black","red"), inset=.04)
}
